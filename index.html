🔗 Google Sheets Integration

Connect your Google Sheets to store all data automatically. Use the Google Apps Script Web App URL.

ℹ️ Deploy a Google Apps Script as a Web App and paste the URL below. See setup instructions at the bottom of this page.
📖 Google Apps Script Setup Guide

Step 1: Go to Google Sheets and create a new spreadsheet. Note the Spreadsheet ID from the URL.

Step 2: Go to Extensions → Apps Script in your spreadsheet.

Step 3: Paste the Apps Script code (provided below) and save it.

Step 4: Click Deploy → New Deployment → Web App. Set "Who has access" to "Anyone". Click Deploy and copy the Web App URL.

Step 5: Paste the Web App URL in the field above and click Save Settings.

// ========== EduTrack Google Apps Script ==========
// Paste this in your Google Apps Script editor

const SHEET_ID = SpreadsheetApp.getActiveSpreadsheet().getId();

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const action = data.action;
    const ss = SpreadsheetApp.openById(SHEET_ID);
    
    if (action === 'addStudent') {
      const sheet = getOrCreateSheet(ss, 'Students');
      if (sheet.getLastRow() === 0) {
        sheet.appendRow(['Roll No','First Name','Last Name','DOB','Gender','Mobile',
          'Email','Address','Course','Batch','Admission Date','Duration',
          'Total Fees','Concession','Guardian','Guardian Mobile','Qualification',
          'ID Type','Remarks','Status']);
      }
      const roll = 'ST' + String(sheet.getLastRow()).padStart(4,'0');
      sheet.appendRow([roll, data.fname, data.lname, data.dob, data.gender,
        data.mobile, data.email, data.address, data.course, data.batch,
        data.admDate, data.duration, data.totalFees, data.concession,
        data.guardian, data.guardianMobile, data.qual, data.idType, data.remarks, 'Active']);
      return output({success:true, rollNo: roll});
    }
    
    if (action === 'getStudents') {
      const sheet = ss.getSheetByName('Students');
      if (!sheet) return output({success:true, data:[]});
      return output({success:true, data: sheetToJson(sheet)});
    }
    
    if (action === 'addPayment') {
      const sheet = getOrCreateSheet(ss, 'Payments');
      if (sheet.getLastRow() === 0) {
        sheet.appendRow(['Receipt No','Date','Roll No','Student Name','Course',
          'Amount','Mode','Reference','Remarks','Created At']);
      }
      const rcpt = 'RCP' + String(Date.now()).slice(-6);
      sheet.appendRow([rcpt, data.date, data.rollNo, data.studentName,
        data.course, data.amount, data.mode, data.reference, data.remarks,
        new Date().toISOString()]);
      return output({success:true, receiptNo: rcpt});
    }
    
    if (action === 'getPayments') {
      const sheet = ss.getSheetByName('Payments');
      if (!sheet) return output({success:true, data:[]});
      return output({success:true, data: sheetToJson(sheet)});
    }
    
    if (action === 'addAttendance') {
      const sheet = getOrCreateSheet(ss, 'Attendance');
      if (sheet.getLastRow() === 0) {
        sheet.appendRow(['Date','Roll No','Student Name','Course','Batch','Status','Topic','Software','Recorded At']);
      }
      data.records.forEach(r => {
        sheet.appendRow([data.date, r.roll, r.name, data.course, data.batch,
          r.status, r.topic || '', r.software || '', new Date().toISOString()]);
      });
      return output({success:true});
    }
    
    if (action === 'getAttendance') {
      const sheet = ss.getSheetByName('Attendance');
      if (!sheet) return output({success:true, data:[]});
      return output({success:true, data: sheetToJson(sheet)});
    }
    
    return output({success:false, error:'Unknown action'});
  } catch(err) {
    return output({success:false, error: err.toString()});
  }
}

function getOrCreateSheet(ss, name) {
  return ss.getSheetByName(name) || ss.insertSheet(name);
}

function sheetToJson(sheet) {
  const [headers, ...rows] = sheet.getDataRange().getValues();
  return rows.map(r => Object.fromEntries(headers.map((h,i) => [h, r[i]])));
}

function output(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  return output({status: 'EduTrack API running'});
}
